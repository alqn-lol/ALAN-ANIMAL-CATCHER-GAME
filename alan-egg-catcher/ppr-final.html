function spawnParticles(x, y, kind = 'sparkle', count = 16) {
  for (let i = 0; i < count; i++) {
    const angle = Math.random() * Math.PI * 2;
    const speed = kind === 'sparkle'
      ? rand(30, 260)
      : rand(80, 380);
    const vx = Math.cos(angle) * speed;
    const vy = Math.sin(angle) * speed *
      (kind === 'sparkle' ? 0.6 : 0.9);
    const size = kind === 'sparkle'
      ? rand(1, 3)
      : rand(3, 8);
    const life = kind === 'sparkle'
      ? rand(0.45, 0.95)
      : rand(0.6, 1.8);
    const color = kind === 'sparkle'
      ? `rgba(${220 + Math.floor(rand(0,20))},${210 + Math.floor(rand(0,20))},${190 + Math.floor(rand(0,20))},1)`
      : `rgba(${140 + Math.floor(rand(0,60))},60,60,1)`;
    state.particles.push({
      x: x || canvas.width / 2,
      y: y || canvas.height - 24,
      vx,
      vy,
      size,
      life,
      age: 0,
      color,
      kind,
    });
  }
}
